import java.util.ArrayList;
import java.util.List;

public class IntegerGenerator {

    /**
     * Menghasilkan daftar bilangan bulat dari nilai awal hingga nilai akhir (inklusif).
     *
     * @param start Nilai awal bilangan bulat, biasanya diatur ke 1.
     * @param end   Nilai akhir bilangan bulat, biasanya diatur ke 100.
     * @return List yang berisi bilangan bulat dari start hingga end (inklusif).
     * @throws IllegalArgumentException jika start lebih besar dari end,
     * atau jika start atau end berada di luar rentang 1 hingga 100.
     */
     // Validasi input: Pastikan start <= end dan keduanya dalam rentang 1-100
    public static List generateIntegers(int start, int end) {
        if (start > end || start < 1 || end > 100) {
            throw new IllegalArgumentException("Nilai start harus <= end dan keduanya dalam rentang 1 hingga 100.");
        }

        // Inisialisasi daftar untuk menyimpan bilangan bulat
        List integers = new ArrayList<>();

        // Loop dari start ke end untuk menambahkan bilangan bulat ke daftar
        for (int i = start; i <= end; i++) {
            integers.add(i);
        }

        return integers;
    }

    /**
     * Metode utama untuk menguji fungsi generateIntegers.
     *
     * @param args Argumen baris perintah (tidak digunakan).
     */
    public static void main(String[] args) {
        try {
            // Menghasilkan dan menampilkan daftar bilangan bulat dari 1 hingga 100
            List integers = generateIntegers(1, 100);
            System.out.println("Bilangan bulat dari 1 hingga 100: " + integers);
        } catch (IllegalArgumentException e) {
            System.err.println("Kesalahan: " + e.getMessage());
        }
    }
}
